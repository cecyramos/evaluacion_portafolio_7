DATOS PARA INGRESAR USANDO LA SHELL

# Ejecutar: python manage.py shell

from catalogo.models import Editorial, Autor, Libro, PerfilAutor
from datetime import date

# Crear Editoriales
e1 = Editorial.objects.create(nombre="Penguin Random House", pais="Estados Unidos", anio_fundacion=1927)
e2 = Editorial.objects.create(nombre="Planeta", pais="España", anio_fundacion=1949)
e3 = Editorial.objects.create(nombre="Alfaguara", pais="España", anio_fundacion=1964)

# Crear Autores
a1 = Autor.objects.create(nombre="Gabriel", apellido="García Márquez", fecha_nacimiento=date(1927,3,6), nacionalidad="Colombiano", biografia="Escritor y periodista colombiano, premio Nobel de Literatura 1982")
a2 = Autor.objects.create(nombre="Isabel", apellido="Allende", fecha_nacimiento=date(1942,8,2), nacionalidad="Chilena", biografia="Escritora chilena, una de las autoras más leídas del mundo")
a3 = Autor.objects.create(nombre="Jorge Luis", apellido="Borges", fecha_nacimiento=date(1899,8,24), nacionalidad="Argentino", biografia="Escritor argentino, uno de los autores más destacados de la literatura del siglo XX")

# Crear Perfiles
PerfilAutor.objects.create(autor=a1, email="gabo@example.com", premios="Premio Nobel de Literatura (1982)")
PerfilAutor.objects.create(autor=a2, email="isabel@example.com", sitio_web="https://www.isabelallende.com", premios="Premio Nacional de Literatura de Chile (2010)")

# Crear Libros
l1 = Libro.objects.create(titulo="Cien años de soledad", isbn="9780060883287", fecha_publicacion=date(1967,6,5), numero_paginas=417, descripcion="Historia de la familia Buendía a lo largo de siete generaciones", editorial=e1)
l1.autores.add(a1)

l2 = Libro.objects.create(titulo="La casa de los espíritus", isbn="9788401242267", fecha_publicacion=date(1982,10,1), numero_paginas=433, descripcion="Saga familiar que abarca tres generaciones en Chile", editorial=e2)
l2.autores.add(a2)

l3 = Libro.objects.create(titulo="El Aleph", isbn="9788420674261", fecha_publicacion=date(1949,6,25), numero_paginas=192, descripcion="Colección de cuentos fantásticos", editorial=e3)
l3.autores.add(a3)

l4 = Libro.objects.create(titulo="El amor en los tiempos del cólera", isbn="9780307389732", fecha_publicacion=date(1985,9,5), numero_paginas=368, descripcion="Historia de amor que dura más de medio siglo", editorial=e1)
l4.autores.add(a1)

l5 = Libro.objects.create(titulo="Paula", isbn="9788401240904", fecha_publicacion=date(1994,1,1), numero_paginas=371, descripcion="Memorias en forma de carta a su hija", editorial=e2)
l5.autores.add(a2)

print("¡Datos poblados exitosamente!")
